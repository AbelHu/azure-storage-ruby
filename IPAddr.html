def validate_ip_and_prefix(ip, cidr)
  if cidr.to_s.empty?
    raise "Cidr is missing for IP '#{ip}'."
  elsif valid?(ip)
    raise "Ip address '#{ip}' is invalid."
  elsif !IPAddr.new(ip).private?
    raise "Ip Address #{ip} must be private."
  end
end
