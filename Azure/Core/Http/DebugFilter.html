def call(req, _next)
  puts "--REQUEST-BEGIN---------------------------"
  puts "method:", req.method, "uri:", req.uri, "headers:", req.headers, "body:", req.body
  puts "--REQUEST-END---------------------------"
  
  r = _next.call
  puts "--RESPONSE-BEGIN---------------------------"
  puts "status_code:", r.status_code, "headers:", r.headers, "body:", r.body
  puts "--RESPONSE-END---------------------------"
  r
end
